74      public void initialize(String []args) {
75          InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
76          if (logger == null) {
77              logger = WaarpInternalLoggerFactory.getLogger(R66ClientGui.class);
78          }
79          if (args.length < 1) {
80              System.err.println("Need client with no database support configuration file as argument");
81              System.exit(2);
82          }
83          if (! FileBasedConfiguration
84                  .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
85              logger
86                      .error("Needs a correct configuration file as first argument");
87              if (DbConstant.admin != null && DbConstant.admin.isConnected) {
88                  DbConstant.admin.close();
89              }
90              ChannelUtils.stopLogger();
91              System.exit(2);
92          }
93          Configuration.configuration.pipelineInit();
94          networkTransaction = new NetworkTransaction();
95      }

113     public boolean checkConnection() {
114         R66Future result = new R66Future(true);
115         TestPacket packet = new TestPacket("MSG", "TestConnection", 100);
116         Message transaction = new Message(networkTransaction, result,
117                 hostId, packet);
118         transaction.run();
119         result.awaitUninterruptibly(30000);
120         if (result.isSuccess()) {
121             R66Result r66result = result.getResult();
122             ValidPacket info = (ValidPacket) r66result.other;
123             GuiResultat = "<html>Test Message    SUCCESS<br>    " +
124                     info.getSheader();
125         } else {
126             GuiResultat = "<html>Test Message    FAILURE<br>    " +
127                     result.getResult().toString();
128         }
129         return result.isSuccess();
130     }

132     public boolean startsTransfer(JProgressBar progressBar, JEditorPane textFieldStatus) {
133         long time1 = System.currentTimeMillis();
134         R66Future future = new R66Future(true);
135         ProgressDirectTransfer transaction = new ProgressDirectTransfer(future, hostId,
136                 filePath, ruleId, information, isMD5, Configuration.configuration.BLOCKSIZE, 
137                 DbConstant.ILLEGALVALUE, networkTransaction, 500,
138                 progressBar, textFieldStatus);
139         logger.debug("Launch transfer: "+hostId+":"+ruleId+":"+filePath);
140         transaction.run();
141         future.awaitUninterruptibly();
142         progressBar.setIndeterminate(true);
143         progressBar.setValue(0);
144         progressBar.setVisible(false);
145         long time2 = System.currentTimeMillis();
146         long delay = time2 - time1;
147         R66Result result = future.getResult();
148         if (future.isSuccess()) {
149             if (result.runner.getErrorInfo() == ErrorCode.Warning) {
150                 GuiResultat = "<html>WARNED<br>    " +
151                         result.runner.toShortNoHtmlString("<br>") +
152                         "<br>    REMOTE: " +
153                         hostId +
154                         (result.file != null? result.file.toString() +
155                                 "" : "no file") + "    delay: " +
156                         delay;
157             } else {
158                 GuiResultat = "<html>SUCCESS<br>    " +
159                     result.runner.toShortNoHtmlString("<br>") +
160                     "<br>    REMOTE: " +
161                     hostId +
162                     (result.file != null? result.file.toString() +
163                             "" : "no file") + "    delay: " +
164                     delay;
165             }
166         } else {
167             if (result == null || result.runner == null) {
168                 GuiResultat = "<html>Transfer in FAILURE with no Id"+
169                     "<br>    REMOTE: " +
170                     hostId + "     "+ future
171                         .getCause().getMessage();
172             } else if (result.runner.getErrorInfo() == ErrorCode.Warning) {
173                 GuiResultat = "<html>Transfer is WARNED<br>    " +
174                     result.runner.toShortNoHtmlString("<br>") +
175                     "<br>    REMOTE: " +
176                     hostId +"    "+ future.getCause().getMessage();
177             } else {
178                 GuiResultat = "<html>Transfer in FAILURE<br>    " +
179                     result.runner.toShortNoHtmlString("<br>")+
180                     "<br>    REMOTE: " +
181                     hostId +"    "+ future.getCause().getMessage();
182             }
183         }
184         return future.isSuccess();
185     }
186     public static String [] getHostIds() {
187         String []results = null;
188         DbHostAuth[] dbHostAuths;
189         try {
190             dbHostAuths = DbHostAuth.getAllHosts(null);
191         } catch (WaarpDatabaseNoConnectionException e) {
192             results = new String[1];
193             results[0] = "NoHostFound";
194             return results;
195         } catch (WaarpDatabaseSqlException e) {
196             results = new String[1];
197             results[0] = "NoHostFound";
198             return results;
199         }
200         if (dbHostAuths.length == 0) {
201             results = new String[1];
202             results[0] = "NoHostFound";
203             return results;
204         }
205         results = new String[dbHostAuths.length];
206         for (int i = 0; i < dbHostAuths.length; i++) {
207             results[i] = dbHostAuths[i].getHostid();
208         }
209         return results;
210     }
