421     public class R66ClientGuiActions extends SwingWorker<String, Integer>{
422         static final int CHECKCONNECTION = 1;
423         static final int STARTTRANSFER = 2;
424         static final int FILESELECT = 3;
425         int method;
426         R66ClientGuiActions(int method) {
427             this.method = method;
428         }
429         /* (non-Javadoc)
430          * @see javax.swing.SwingWorker#doInBackground()
431          */
432         @Override
433         protected String doInBackground() throws Exception {
434             disableAllButtons();
435             startRequest();
436             switch (method) {
437                 case CHECKCONNECTION:
438                     checkConnection();
439                     break;
440                 case STARTTRANSFER:
441                     startTransfer();
442                     break;
443                 case FILESELECT:
444                     findFile();
445                     break;
446                 default:
447                     environnement.GuiResultat = "Action not recognized";
448             }
449             setStatus(environnement.GuiResultat);
450             if (method != FILESELECT) {
451                 showDialog();
452             } else {
453                 enableAllButtons();
454             }
455             stopRequest();
456             return environnement.GuiResultat;
457         }
458     }
459     
460     private void showDialog() {
461         disableAllButtons();
462         if (dialog != null) {
463             dialog.dispose();
464             dialog = null;
465         }
466         if (dialog == null) {
467             dialog = new R66Dialog();
468             dialog.setLocationRelativeTo(frmRClientGui);
469             if (dialog.isAlwaysOnTopSupported()) {
470                 dialog.setAlwaysOnTop(true);
471             } else {
472                 dialog.toFront();
473             }
474         }
475         dialog.textPaneDialog.setText(environnement.GuiResultat);
476         dialog.setVisible(true);
477         dialog.requestFocus();
478     }
479     private void setStatus(String mesg) {
480         textFieldStatus.setText(mesg);
481     }
482     private void startRequest() {
483         progressBarTransfer.setIndeterminate(true);
484         progressBarTransfer.setValue(0);
485         progressBarTransfer.setVisible(true);
486         textPaneLog.setText("");
487     }
488     private void stopRequest() {
489         progressBarTransfer.setIndeterminate(true);
490         progressBarTransfer.setValue(0);
491         progressBarTransfer.setVisible(false);
492         frmRClientGui.toFront();
493         frmRClientGui.requestFocus();
494     }
495     private void checkConnection() {
496         startRequest();
497         disableAllButtons();
498         environnement.hostId = (String) comboBoxHosts.getSelectedItem();
499         if (environnement.hostId == null || environnement.hostId.trim().length() <= 0) {
500             environnement.hostId = "NO HOST";
501             environnement.GuiResultat = "No Host specified!";
502         } else {
503             environnement.hostId = environnement.hostId.trim();
504             environnement.checkConnection();
505         }
506         setStatus(environnement.GuiResultat);
507         showDialog();
508         stopRequest();
509     }
510     private void startTransfer() {
511     	logger.debug("start startTransfer");
512         disableAllButtons();
513         environnement.hostId = (String) comboBoxHosts.getSelectedItem();
514         environnement.ruleId = (String) comboBoxRules.getSelectedItem();
515         environnement.filePath = textFieldFile.getText();
516         environnement.information = textFieldInformation.getText();
517         environnement.isMD5 = checkBoxMD5.isSelected();
518         
519         boolean ok = true;
520         if (environnement.hostId == null || environnement.hostId.trim().length() <= 0) {
521             environnement.hostId = "NO HOST";
522             ok = false;
523         } else {
524             environnement.hostId = environnement.hostId.trim();
525         }
526         if (environnement.filePath == null || environnement.filePath.trim().length() <= 0) {
527             environnement.filePath = "NO FILE";
528             ok = false;
529         } else {
530             environnement.filePath = environnement.filePath.trim();
531         }
532         if (environnement.information == null || environnement.information.trim().length() <= 0) {
533             environnement.information = "";
534         } else {
535             environnement.information = environnement.information.trim();
536         }
537         if (environnement.ruleId == null || environnement.ruleId.trim().length() <= 0) {
538             environnement.ruleId = "NO RULE";
539             ok = false;
540         } else {
541             environnement.ruleId = environnement.ruleId.trim();
542         }
543         if (ok) {
544         	logger.debug("start startTransfer: "+environnement.toString());
545             environnement.startsTransfer(progressBarTransfer, textFieldStatus);
546         } else {
547             environnement.GuiResultat = "<html>Not enough arg to start the transfer:<br>   "+
548                 "HostId: "+environnement.hostId+" Rule: "+environnement.ruleId+
549                 " File: "+environnement.filePath;
550         }
551         setStatus(environnement.GuiResultat);
552         showDialog();
553     }
554     private void findFile() {
555         startRequest();
556         disableAllButtons();
557         try {
558             JFileChooser fc = new JFileChooser();
559             int returnVal = fc.showOpenDialog(frmRClientGui);
560             if (returnVal == JFileChooser.APPROVE_OPTION) {
561                 File file = fc.getSelectedFile();
562                 try {
563 					textFieldFile.setText(file.getCanonicalPath());
564 				} catch (IOException e) {
565 				}
566                 setFindFile();
567                 environnement.GuiResultat = "New file sets";
568             }
569         } finally {
570             enableAllButtons();
571             stopRequest();
572         }
573     }
574     private void setFindFile() {
575         String text = null;
576         try {
577             text = textFieldFile.getText();
578         } catch (NullPointerException e1) {
579             text = null;
580         }
581         if (text != null) {
582     		File file = new File(text);
583     		if (file.exists()) {
584 				text = file.toURI().toString();
585     		} else {
586     			//text = "unknown file";
587     		}
588             textFieldFile.setText(text);
589         }
590     }
